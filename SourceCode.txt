function onSampleData(event:SampleDataEvent):void{
	var bArray:ByteArray = event.data;
	var i:int;
	//受け取ったサウンドデータの絶対値の総和を求める
	for( i = 0 ; i < bArray.bytesAvailable ; i++) {
		sample += Math.abs(bArray.readFloat());
	}
	sample = sample / bArray.bytesAvailable * 1000; //平均値を変えて100倍
	trace( sample );
	if( sample >= 100 ){
		nobiritu = 7/4;
	} else {
		if( sample >= 90){
			nobiritu = 6/4;
		} else {
			if( sample >= 80){
				nobiritu = 5/4;
			} else {
				if( sample >= 70){
					nobiritu = 3/4;
				} else {
					if( sample >= 60){
						nobiritu = 2/4;
					} else {
						if( sample >= 50){
							nobiritu = 1/4;
						} else {
							nobiritu = 0;
						}
					}
				}
			}
		}
	}
}

function nobiru(event:Event) {
	kuki.height += nobiritu;
	ohana.y -= nobiritu ;
	ohana.rotation += 1;
	ohanaItiY = ohana.y;
	kukiHeight = kuki.height;
	if( ohanaItiY <= 300 ){
		trace(kuki.height);
		angou1.x = 49;
		angou1.y = 138.70;
		nobiritu = 0;
		mic.removeEventListener(SampleDataEvent.SAMPLE_DATA, onSampleData);
		if( hanaFlag == false ){
			PSEseikai.play(0,0);
			hanaFlag = true;
		}
		ohanaItiY = ohana.y;
		kukiHeight = kuki.height;
	}	
}

function onSwipe4(ev:TransformGestureEvent):void{
	if(ev.offsetX == 1){
		stage.removeEventListener(TransformGestureEvent.GESTURE_SWIPE,onSwipe4);
		accelerometer.removeEventListener(AccelerometerEvent.UPDATE,onAccUpdate);
		stage.removeEventListener(Event.ENTER_FRAME, Slider);
		gotoAndStop("play1");	
	}
	if(ev.offsetX == -1){
		stage.removeEventListener(TransformGestureEvent.GESTURE_SWIPE,onSwipe4);
		accelerometer.removeEventListener(AccelerometerEvent.UPDATE,onAccUpdate);
		stage.removeEventListener(Event.ENTER_FRAME, Slider);
		gotoAndStop("play3");
	}
}

if( Accelerometer.isSupported ) {
	var accelerometer:Accelerometer = new Accelerometer();
	accelerometer.addEventListener(AccelerometerEvent.UPDATE,onAccUpdate);
}

function onAccUpdate(event:AccelerometerEvent):void{
	ItiX = event.accelerationX;
	ItiY = event.accelerationY;
	ItiZ = event.accelerationZ;
}

function Slider(event:Event):void {
	//赤球
	if( 0 < ItiZ ) {
		if( 0.6 < ItiY && ItiY < 1){
			mcRedS.y += 1 * RPM;
		} else if ( 0.3 < ItiY && ItiY < 0.6 ){
			mcRedS.y += 2 * RPM;
		} else if ( 0 < ItiY && ItiY < 0.3 ){
			mcRedS.y += 2 * RPM;
		}	
	}
	if( Upkabe.hitTestObject(mcRedS)){
		RPM = 1;
	} else if (Downkabe.hitTestObject(mcRedS)){
		RPM = -1;
	}
	//ピンク球
	if( 0.7 < ItiZ && ItiZ < 1) {
		if( -0.2 < ItiY && ItiY < 0){
			mcPinkS.y += 2 * PPM;
		} else if ( -0.4 < ItiY && ItiY < -0.2 ){
			mcPinkS.y += 4 * PPM;
		} else if ( -0.7 < ItiY && ItiY < -0.4 ){
			mcPinkS.y += 6 * PPM;
		}	
	}
	if( Upkabe.hitTestObject(mcPinkS)){
		PPM = 1;
	} else if (Downkabe.hitTestObject(mcPinkS)){
		PPM = -1;
	}
	//青球
	if( 0 < ItiX ) {
		if( 1 > ItiZ && ItiZ > 0.7){
			mcBlueS.y += 3 * BPM;
		} else if ( 0.7 > ItiZ && ItiZ > 0.4 ){
			mcBlueS.y += 6 * BPM;
		} else if ( 0.4 > ItiZ && ItiZ > 0 ){
			mcBlueS.y += 9 * BPM;
		}	
	}
	if( Upkabe.hitTestObject(mcBlueS)){
		BPM = 1;
	} else if (Downkabe.hitTestObject(mcBlueS)){
		BPM = -1;
	}
	//緑球
	if( ItiX < 0 ) {
		if( 0 < ItiZ && ItiZ < 0.3 ){
			mcGreenS.y += 3.5 * GPM;
		} else if ( 0.3 < ItiZ && ItiZ < 0.6 ){
			mcGreenS.y += 7 * GPM;
		} else if ( 0.6 < ItiZ && ItiZ < 1 ){
			mcGreenS.y += 10.5 * GPM;
		}	
	}
	if( Upkabe.hitTestObject(mcGreenS)){
		GPM = 1;
	} else if (Downkabe.hitTestObject(mcGreenS)){
		GPM = -1;
	}
	//ヒント開示
	if( mcRrod.hitTestObject(mcRedS) && mcProd.hitTestObject(mcPinkS) && mcBrod.hitTestObject(mcBlueS) && mcGrod.hitTestObject(mcGreenS)){
		stage.removeEventListener(Event.ENTER_FRAME, Slider);
		mcHintGreen.visible = true;
		PSEseikai.play(0,0);
	}
}
